## 三、统一响应格式

所有 API 统一返回格式：

```json
{
  "code": 0,
  "message": "success",
  "data": {}
}
```

| 字段 | 类型 | 说明 |
|------|------|------|
| code | int | 状态码，0 表示成功，非 0 表示失败 |
| message | string | 消息描述 |
| data | object | 返回数据 |

**常用状态码**：

| code | 说明 |
|------|------|
| 0 | 成功 |
| 400 | 请求参数错误 |
| 401 | 未认证 / token 无效 |
| 403 | 无权限 |
| 404 | 资源不存在 |
| 500 | 服务器内部错误 |

**统一响应工具函数**：
```python
# utils/response.py
from flask import jsonify

def success(data=None, message='success'):
    return jsonify({
        'code': 0,
        'message': message,
        'data': data
    }), 200

def error(code=500, message='error', data=None):
    return jsonify({
        'code': code,
        'message': message,
        'data': data
    }), 200
```

### 6.1 YOLO 检测服务

```python
# services/yolo_service.py
import os
import time
import uuid
import cv2
from ultralytics import YOLO
from flask import current_app

class YoloService:
    _instance = None
    _model = None
    _model_loaded = False

    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
        return cls._instance

    def load_model(self, model_path=None):
        """加载模型"""
        if model_path is None:
            model_path = current_app.config['YOLO_MODEL_PATH']

        if not os.path.exists(model_path):
            raise FileNotFoundError(f"模型文件不存在: {model_path}")

        self._model = YOLO(model_path)
        self._model_loaded = True
        return self._model

    def is_model_loaded(self):
        return self._model_loaded and self._model is not None

    def get_class_names(self):
        """获取类别名称列表"""
        if not self.is_model_loaded():
            return []
        return list(self._model.names.values())

    def detect_image(self, image_path, result_dir, save_result=True):
        """检测单张图片"""
        if not self.is_model_loaded():
            self.load_model()

        start_time = time.time()
        results = self._model(
            image_path,
            conf=current_app.config['YOLO_CONF_THRESHOLD'],
            iou=current_app.config['YOLO_IOU_THRESHOLD']
        )
        result = results[0]

        processing_time = time.time() - start_time

        # 保存结果图
        result_path = None
        if save_result:
            os.makedirs(result_dir, exist_ok=True)
            filename = f"{uuid.uuid4().hex}_result.jpg"
            result_path = os.path.join(result_dir, filename)
            result_img = result.plot()
            cv2.imwrite(result_path, cv2.cvtColor(result_img, cv2.COLOR_RGB2BGR))

        # 解析检测结果
        detections = []
        class_summary = {}
        for i in range(len(result.boxes)):
            cls_id = int(result.boxes.cls[i])
            cls_name = result.names[cls_id]
            conf = float(result.boxes.conf[i])
            bbox = result.boxes.xyxy[i].tolist()

            detections.append({
                'class_id': cls_id,
                'class_name': cls_name,
                'confidence': round(conf, 4),
                'bbox': [int(x) for x in bbox]
            })
            class_summary[cls_name] = class_summary.get(cls_name, 0) + 1

        return {
            'detections': detections,
            'total_count': len(detections),
            'class_summary': class_summary,
            'result_path': result_path,
            'processing_time': round(processing_time, 3),
            'model_version': 'yolo26s'
        }

    def detect_video(self, video_path, result_dir, save_result=True, progress_callback=None):
        """检测视频（可选实现）"""
        pass

yolo_service = YoloService()
```

### 6.2 LLM 问答服务

```python
# services/llm_service.py
import os
from flask import current_app

class LLMService:
    _instance = None
    _client = None

    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
        return cls._instance

    def _get_client(self):
        """获取 LLM 客户端"""
        if self._client is None:
            api_key = current_app.config['LLM_API_KEY']
            base_url = current_app.config['LLM_BASE_URL']
            # 这里根据实际使用的大模型初始化客户端
            # 示例：通义千问
            from openai import OpenAI
            self._client = OpenAI(api_key=api_key, base_url=base_url)
        return self._client

    def generate_answer(self, question, knowledge_list=None, detection_context=''):
        """生成回答"""
        client = self._get_client()

        # 构建提示词
        knowledge_text = '\n'.join(knowledge_list) if knowledge_list else ''
        context_parts = []
        if knowledge_text:
            context_parts.append(f"参考知识：\n{knowledge_text}")
        if detection_context:
            context_parts.append(f"检测结果：{detection_context}")

        system_prompt = f"""你是一个专业的知识助手，请根据参考知识回答用户的问题。
回答要准确、专业，尽量结合提供的参考知识。
如果参考知识中没有相关内容，请如实说明。
{chr(10).join(context_parts)}
"""

        response = client.chat.completions.create(
            model=current_app.config['LLM_MODEL'],
            messages=[
                {"role": "system", "content": system_prompt},
                {"role": "user", "content": question}
            ],
            temperature=0.7,
            max_tokens=1000
        )

        answer = response.choices[0].message.content
        sources = [k[:50] + '...' for k in knowledge_list] if knowledge_list else []

        return {
            'answer': answer,
            'sources': sources
        }

llm_service = LLMService()
```

### 6.3 知识库服务

```python
# services/knowledge_service.py
import chromadb
from flask import current_app
from models import Knowledge, db

class KnowledgeService:
    _instance = None
    _collection = None

    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
        return cls._instance

    def _get_collection(self):
        """获取向量集合"""
        if self._collection is None:
            chroma_path = current_app.config['CHROMA_PATH']
            collection_name = current_app.config['CHROMA_COLLECTION']
            client = chromadb.PersistentClient(path=chroma_path)
            self._collection = client.get_or_create_collection(name=collection_name)
        return self._collection

    def search(self, query, top_k=3):
        """搜索相关知识"""
        collection = self._get_collection()
        results = collection.query(query_texts=[query], n_results=top_k)

        knowledge_list = []
        for i in range(len(results['documents'][0])):
            knowledge_list.append(results['documents'][0][i])

        return knowledge_list

    def add_knowledge(self, title, content, category=None, source=None):
        """添加知识条目"""
        # 保存到数据库
        kb = Knowledge(
            title=title,
            content=content,
            category=category,
            source=source
        )
        db.session.add(kb)
        db.session.flush()

        # 添加到向量库
        collection = self._get_collection()
        doc_id = f"kb_{kb.id}"
        collection.add(
            documents=[content],
            ids=[doc_id],
            metadatas=[{"title": title, "category": category}]
        )
        kb.vector_id = doc_id
        db.session.commit()

        return kb

    def sync_all_to_vector(self):
        """将所有知识库条目同步到向量库"""
        collection = self._get_collection()
        all_knowledge = Knowledge.query.filter_by(is_active=True).all()

        documents = []
        ids = []
        metadatas = []

        for kb in all_knowledge:
            doc_id = f"kb_{kb.id}"
            documents.append(kb.content)
            ids.append(doc_id)
            metadatas.append({"title": kb.title, "category": kb.category})
            kb.vector_id = doc_id

        if documents:
            collection.upsert(documents=documents, ids=ids, metadatas=metadatas)
            db.session.commit()

        return len(all_knowledge)

knowledge_service = KnowledgeService()
```

### 7.1 认证中间件

使用 Flask-JWT-Extended 实现 JWT 认证。

```python
# middleware/auth_middleware.py
from functools import wraps
from flask_jwt_extended import get_jwt_identity
from models import User
from utils.response import error

def login_required(fn):
    @wraps(fn)
    def wrapper(*args, **kwargs):
        try:
            user_id = get_jwt_identity()
            if not user_id:
                return error(code=401, message='请先登录')

            user = User.query.get(user_id)
            if not user or not user.is_active:
                return error(code=401, message='用户不存在或已禁用')

            kwargs['current_user'] = user
            return fn(*args, **kwargs)
        except Exception as e:
            return error(code=401, message='Token无效或已过期')

    return wrapper

def admin_required(fn):
    @wraps(fn)
    def wrapper(*args, **kwargs):
        user = kwargs.get('current_user')
        if not user or user.role != 'admin':
            return error(code=403, message='需要管理员权限')
        return fn(*args, **kwargs)
    return login_required(wrapper)
```

### 8.1 文件工具

```python
# utils/file_utils.py
import os
import uuid
from werkzeug.utils import secure_filename

def allowed_file(filename, allowed_extensions):
    """检查文件扩展名是否允许"""
    return '.' in filename and \
           filename.rsplit('.', 1)[1].lower() in allowed_extensions

def save_uploaded_file(file, upload_dir, prefix=''):
    """保存上传的文件，返回保存后的路径和文件名"""
    os.makedirs(upload_dir, exist_ok=True)

    # 生成唯一文件名
    original_name = secure_filename(file.filename)
    ext = original_name.rsplit('.', 1)[1].lower() if '.' in original_name else ''
    new_filename = f"{prefix}{uuid.uuid4().hex}.{ext}" if ext else f"{prefix}{uuid.uuid4().hex}"

    save_path = os.path.join(upload_dir, new_filename)
    file.save(save_path)

    return save_path, new_filename, original_name

def get_file_size(file_path):
    """获取文件大小（字节）"""
    return os.path.getsize(file_path) if os.path.exists(file_path) else 0
```
