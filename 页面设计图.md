### 1.3 首页状态图

```mermaid
stateDiagram-v2
    [*] --> EmptyState: 首次进入<br/>无设备组无设备
    
    EmptyState --> NormalState: 添加设备或创建设备组
    
    state NormalState {
        [*] --> GroupCollapsed: 默认收起
        GroupCollapsed --> GroupExpanded: 点击组标题
        GroupExpanded --> GroupCollapsed: 再次点击组标题
        
        state GroupExpanded {
            [*] --> HasDevices: 有设备
            HasDevices --> Dragging: 长按1秒+移动>20px
            Dragging --> HasDevices: 手指抬起
            HasDevices --> Scanning: 点击添加设备
            Scanning --> HasDevices: 扫描完成/添加设备
        }
        
        GroupExpanded --> EmptyGroup: 组内无设备
    end
    
    NormalState --> OfflineState: 设备离线
    OfflineState --> NormalState: 设备上线
    
    NormalState --> AlarmTriggered: 点击测试预警
    state AlarmTriggered {
        [*] --> SelectDevice: 选择设备
        SelectDevice --> SelectAlarmType: 选择异常类型
        SelectAlarmType --> NavigateToControl: 跳转控制页
    }
    
    NormalState --> [*]: 退出APP
```

### 2.3 控制页面状态图

```mermaid
stateDiagram-v2
    [*] --> EmptyState: 未选中设备
    
    EmptyState --> NormalState: 从首页点击设备卡片进入
    
    state NormalState {
        [*] --> PowerOff: 设备已关机 P0
        PowerOff --> PowerOn: 点击开机 >AFO
        PowerOn --> PowerOff: 点击关机 >AFF
        
        state PowerOn {
            [*] --> Level1_4: P1-P4 蓝色
            Level1_4 --> Level5_8: 加档 P5-P8 橙色
            Level5_8 --> Level9_12: 加档 P9-P12 红色
            Level9_12 --> Level5_8: 减档
            Level5_8 --> Level1_4: 减档
        }
        
        PowerOn --> TimerRunning: 设置定时
        state TimerRunning {
            [*] --> Counting: 倒计时运行中
            Counting --> Paused: 点击暂停
            Paused --> Counting: 点击继续
            Counting --> TimerEnd: 倒计时归零
            TimerEnd --> PowerOff: 自动关机 >AFF
        }
        TimerRunning --> PowerOn: 取消定时
    }
    
    NormalState --> OfflineState: 设备断开连接
    OfflineState --> NormalState: 设备重新连接
    
    NormalState --> AlarmState: 收到异常上报 >AE1~>AE6
    state AlarmState {
        [*] --> AlarmActive: 警报激活
        AlarmActive --> Vibrating: 持续震动
        Vibrating --> AlarmConfirmed: 点击关机确认
        AlarmConfirmed --> PowerOff: 发送 >AFF 关机
    }
    
    AlarmState --> NormalState: 所有警报已确认关闭
    
    NormalState --> [*]: 返回首页
```

## 三、页面间数据流总图

```mermaid
flowchart LR
    subgraph "首页 index.vue"
        DeviceGroups["设备组数据<br/>deviceGroups"]
        SelectedDevice["选中设备ID<br/>selectedDeviceId"]
        AlarmData["警报数据<br/>pendingAlarm"]
    end

    subgraph "数据传递方式"
        Storage["uni.setStorageSync"]
        Event["uni.$emit / uni.$on"]
        Navigate["uni.navigateTo"]
    end

    subgraph "控制页面 detail.vue"
        DeviceInfo["设备信息<br/>从Storage读取"]
        AlarmDisplay["警报显示<br/>从Storage/Event接收"]
        CommandLogs["指令日志<br/>页面内维护"]
    end

    subgraph "蓝牙模块 ble.js"
        BLEManager["BLEManager 单例"]
        CMD["指令常量 CMD<br/>完整指令集"]
        Send["sendCommand()"]
        Receive["onMessage()"]
    end

    DeviceGroups -->|点击卡片| SelectedDevice -->|setStorageSync| Storage -->|getStorageSync| DeviceInfo
    AlarmData -->|setStorageSync| Storage -->|getStorageSync| AlarmDisplay
    AlarmData -->|$emit| Event -->|$on| AlarmDisplay
    SelectedDevice -->|navigateTo| Navigate
    
    DeviceInfo -->|连接设备| BLEManager
    BLEManager -->|sendCommand| Send -->|蓝牙写入| 设备
    设备 -->|蓝牙通知| Receive -->|parseResponse| AlarmDisplay
    Receive -->|parseResponse| CommandLogs
```
